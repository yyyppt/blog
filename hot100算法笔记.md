## 和为k的子数组

用preSum表示开头到当前的累积和，要求的是preSum[i] - preSum[j] = k即preSum[j] = preSum[i] - k，所以我们在遍历时就看之前有没有出现preSum - k，并把记录的count数量++

> 记得初始化mp[0] = 1

```c++
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int preSum = 0;
        int count = 0;
        unordered_map<int, int> mp;
        mp[0] = 1;
        for (int i = 0; i < nums.size(); i++) {
            preSum += nums[i];
            if (mp.find(preSum - k) != mp.end()) {
                count += mp[preSum - k];
            }
            mp[preSum]++;
        }
        return count;
    }
};
```

## 滑动窗口最大值

使用一个双端队列进行滑窗的维护，在for循环里的前面先把新的第i个元素和队尾下标的元素比较，若是比队尾的大则pop弹出队尾的，然后给队尾插入数组的下标，若是队头下标已经小于等于i - k，即队列已满，则弹出队头

最后从i >= k - 1时开始用数组存下所有队头

```c++
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        deque<int> dq;
        vector<int> res;
        for (int i = 0; i < nums.size(); i++) {
            while (!dq.empty() && nums[i] > nums[dq.back()]) {
                dq.pop_back();
            }
            dq.push_back(i);
            if (!dq.empty() && dq.front() <= i - k) {
                dq.pop_front();
            }
            if (i >= k - 1) {
                res.push_back(nums[dq.front()]);
            }
        }
        return res;
    }
};
```

## 最小覆盖子串

这里要两个哈希表分别记录滑窗时窗口里的字符，和t字符串里我们要找的need字符

先把所有的要找的字符传进need表里need[c]++

然后就开始有个left和right来进行窗口的滑动，while(right<s.size())先记录每个字符，然后right++,若是need表里有当前字符，则window[c]++,然后对比窗口的和need里的，若是相等则有一个专门记录相同字符数的needSum++

当needSum==need.size()的时候，即滑窗和需要的字符数想等了，然后就若是right - left < minLen（这是提前定义好的无穷大INT_MAX）则说明需要更新起始点和长度了，那么有个起始点start = left，minLen = right - left

然后就可以在里面缩小滑窗范围了，char d = s[left]，left++

若need.count(d)，即缩小的有need里的，那末若是window[d] == need[d]，needSum--,然后滑窗里的d--

最后若是minlen == INT_MAX返回""，否则返回s.substr(start, minlen)即s里的那一段最短的符合条件的字符

```c++
class Solution {
public:
    string minWindow(string s, string t) {
        unordered_map<char, int> need;
        unordered_map<char, int> window;
        for (char c : t) {
            need[c]++;
        }
        int left = 0, right = 0;
        int needSum = 0;
        int minLen = INT_MAX, start = 0;
        while (right < s.size()) {
            char c = s[right];
            right++;
            if (need.count(c)) {
                window[c]++;
                if (window[c] == need[c]) {//这里可以防止window里的字符超出
                    needSum++;//这个记录的是总共有多少种的字符，比如aabc只有三个need.size()
                }
            }
            while (needSum == need.size()) {
                if (right - left < minLen) {
                        start = left;
                        minLen = right - left;
                }
                char d = s[left];
                left++;
                if (need.count(d)) {
                    if (window[d] == need[d]) {
                        needSum--;
                    }
                    window[d]--;
                }
            }
        }
        return minLen == INT_MAX ? "" : s.substr(start, minLen);
    }
};
```

## 最大子数组和

这里主要有个前缀和与贪心的思想，如果前面的总和是负数，那么它对后面的任何子数组都是负贡献,则重新开始记，然后每次更新最大值就行

- 时间复杂度：O(n)
- 空间复杂度：O(1)

```c++
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int maxSum = INT_MIN;
        int sum = 0;
        for (int i = 0; i < nums.size(); i++) {
            if (sum < 0) {
                sum = 0;
            }
            sum += nums[i];
            maxSum = max(maxSum, sum);
        }
        return maxSum;
    }
};
```

这里还有一个分治的解法，能把复杂度精确到

- 时间复杂度：O(n log n)
- 空间复杂度：O(log n)

> 对于任意区间 [l, r]，最大子数组和只可能出现在左半边、右半边，或者跨越中点。前两种情况递归求解，第三种情况通过“左边最大后缀和 + 右边最大前缀和”得到。最后取三者最大值即可。

```c++
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
       return divide(nums, 0, nums.size() - 1); 
    }
    int divide(vector<int>& nums, int left, int right) {
        if (left == right) {
            return nums[left];
        }
        int mid = left + (right - left) / 2;
        int leftMax = divide(nums, left, mid);
        int rightMax = divide(nums, mid + 1, right);
        int crossMax = crossSum(nums, left, mid, right);
        return max({leftMax, rightMax, crossMax});
    }
    int crossSum(vector<int>& nums, int left, int mid, int right) {
        int leftSum = INT_MIN;
        int sum = 0;
        for (int i = mid; i >= left; i--) {
            sum += nums[i];
            leftSum = max(sum, leftSum);
        }
        int rightSum = INT_MIN;
        sum = 0;
        for (int i = mid + 1; i <= right; i++) {
            sum += nums[i];
            rightSum = max(sum, rightSum);
        }
        return leftSum + rightSum;
    }
};
```

## 合并区间

先按起点排序，然后遍历区间，如果当前区间和结果集最后一个区间有重叠就合并，否则直接加入结果

这里注意重叠时比较的是if (intervals[i] [0] <= last[1])，这里是因为判断的是重叠的部分，要用存的最后面比较新的最前面

```c++
vector<vector<int>> merge(vector<vector<int>>& intervals) {
    if (intervals.empty()) return {};
    
    sort(intervals.begin(), intervals.end());
    
    vector<vector<int>> res;
    res.push_back(intervals[0]);
    for (int i = 1; i < intervals.size(); i++) {
        // 当前结果中的最后一个区间
        vector<int>& last = res.back();
        // 有重叠
        if (intervals[i][0] <= last[1]) {
            last[1] = max(last[1], intervals[i][1]);
        } else {
            // 无重叠
            res.push_back(intervals[i]);
        }
    }
    
    return res;
}
```

   ```c++
class Solution {

        int n = nums.size();
        vector<int> res(n);
   ```

```c++
class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int n = nums.size();
        // 第一遍：把每个数放到它应该在的位置,交换判断的的都是nums[nums[i] - 1]
        for (int i = 0; i < n; i++) {

        for (int i = 0; i < n; i++) {
            if (nums[i] != i + 1) {
                return i + 1;
            }
        }
        //如果1~n都在，答案就是 n+1
        return n + 1;
    }
```

```c++
            }
        }
        for (int i = 0; i < m; i++) {
            if (matrix[0][i] == 0) {
                colZero = true;
            }
        }

                if (matrix[i][j] == 0) {
                    matrix[i][0] = 0;
                    matrix[0][j] = 0;
                }
            }
        }
        for (int i = 1; i < n; i++) {

                if (matrix[i][0] == 0 || matrix[0][j] == 0) {
                    matrix[i][j] = 0;
                }
            }

            for (int i = 0; i < n; i++) {
                matrix[i][0] = 0;
            }
            
        }

            for (int j = 0; j < m; j++) {
                matrix[0][j] = 0;
            }
        }
    }
```

## 螺旋矩阵

这题其实就是模拟也没啥特别要注意的，就是要注意在最后的从底下右到左和左边下到上的时候要进行防止模拟跳出情况进行top和left的判断

```c++
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        vector<int> result;
        
        int rows = matrix.size();
        int cols = matrix[0].size();
        
        int top = 0;
        int bottom = rows - 1;
        int left = 0;
        int right = cols - 1;
        
        while (left <= right && top <= bottom) {
            // 1. 从左到右
            for (int i = left; i <= right; i++) {
                result.push_back(matrix[top][i]);
            }
            top++;
            
            // 2. 从上到下
            for (int i = top; i <= bottom; i++) {
                result.push_back(matrix[i][right]);
            }
            right--;
            // 3. 从右到左
            if (top <= bottom) {
                for (int i = right; i >= left; i--) {
                    result.push_back(matrix[bottom][i]);
                }
                bottom--;
            }
            // 4. 从下到上
            if (left <= right) {
                for (int i = bottom; i >= top; i--) {
                    result.push_back(matrix[i][left]);
                }
                left++;
            }
        }
        
        return result;
    }
};
```

```c++
    bool searchMatrix(vector<vector<int>>& matrix, int target) {

            return false;
        }
        int rows = matrix.size();
        int cols = matrix[0].size();
        int i = 0, j = cols - 1;
```

```c++
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode dum(0);
        ListNode* cur = &dum;
        int carry = 0;
        while (l1 || l2 || carry) {
            int sum = carry;
            if (l1) {
                sum += l1->val;
                l1 = l1->next;
            }
            if (l2) {
                sum += l2->val;
                l2 = l2->next;

            cur = cur->next;
            carry = sum / 10;
```

## 排序链表

时间复杂度O(nlogn)空间复杂度O(logn)

```c++
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        if (head == nullptr || head->next == nullptr) {
            return head;
        }
        // 1. 快慢指针找中点
        ListNode* slow = head;
        ListNode* fast = head->next;

        while (fast != nullptr && fast->next != nullptr) {
            slow = slow->next;
            fast = fast->next->next;
        }

        // 2. 从中间断开链表
        ListNode* rightHead = slow->next;
        slow->next = nullptr;

        // 3. 分别排序左右两边
        ListNode* left = sortList(head);
        ListNode* right = sortList(rightHead);

        // 4. 合并两个有序链表
        return merge(left, right);
    }

    ListNode* merge(ListNode* l1, ListNode* l2) {
        ListNode dummy(0);
        ListNode* cur = &dummy;

        while (l1 != nullptr && l2 != nullptr) {
            if (l1->val <= l2->val) {
                cur->next = l1;
                l1 = l1->next;
            } else {
                cur->next = l2;
                l2 = l2->next;
            }
            cur = cur->next;
        }

        if (l1 != nullptr) {
            cur->next = l1;
        } else {
            cur->next = l2;
        }

        return dummy.next;
    }
};
```

## 二叉树的右视图

层序遍历：

这里有个易错点就是必须先左入队再右入队

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> rightSideView(TreeNode* root) {
        queue<TreeNode *> que;
        vector<int> vec;
        if (root != nullptr) {
            que.push(root);
        }
        while (!que.empty()) {
            int size = que.size();           
            for (int i = 0; i < size; i++) {
                TreeNode* node = que.front();
                que.pop();
                if (i == size - 1) {
                    vec.push_back(node->val); 
                }
                if (node->left) {
                    que.push(node->left);
                }
                if (node->right) {
                    que.push(node->right);
                }
            }
        }
        return vec;
    }
};
```

```c++
class Solution {
public:
    vector<int> result;

    void dfs(TreeNode* node, int depth) {
        if (node == nullptr) return;

        // 如果当前深度还没有记录，说明是这一层第一个节点
        if (depth == result.size()) {
            result.push_back(node->val);
        }
        // 先右再左（关键）
        dfs(node->right, depth + 1);
        dfs(node->left, depth + 1);
    }

    vector<int> rightSideView(TreeNode* root) {
        dfs(root, 0);
        return result;
    }
};
```

## 二叉树的直径

```c++
class Solution {
public:
    int res = 0;
    int diameterOfBinaryTree(TreeNode* root) {
        depth(root);
        return res;
    }
    int depth(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }
        int leftDepth = depth(root->left);
        int rightDepth = depth(root->right);
        res = max(res, leftDepth + rightDepth);
        return max(leftDepth, rightDepth) + 1;
    }
};
```
